# Estágio 1: Build da Aplicação
FROM node:18-alpine AS builder
WORKDIR /app

# Copia o package.json e o lock file
COPY package*.json ./

# Instala as dependências
RUN npm install

# Copia o restante do código da aplicação
COPY . .

# Copia as variáveis de ambiente de build (se houver alguma)
# As variáveis de runtime serão injetadas pelo Docker Compose
ARG NEXT_PUBLIC_SUPABASE_URL
ARG NEXT_PUBLIC_SUPABASE_ANON_KEY
ENV NEXT_PUBLIC_SUPABASE_URL=${NEXT_PUBLIC_SUPABASE_URL}
ENV NEXT_PUBLIC_SUPABASE_ANON_KEY=${NEXT_PUBLIC_SUPABASE_ANON_KEY}

# Roda o build do Next.js
RUN npm run build

# Estágio 2: Produção - Imagem final e leve
FROM node:18-alpine
WORKDIR /app

# Copia os artefatos do build do estágio anterior
COPY --from=builder /app/.next ./.next
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package.json ./package.json
COPY --from=builder /app/public ./public

# Expõe a porta que o Next.js usa
EXPOSE 3000

# Comando para iniciar a aplicação em produção
CMD ["npm", "start"]
